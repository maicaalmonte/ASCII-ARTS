24   # Function to fill the face outline with song titles
25   def fill_face_with_songs(outline, song_list):
26       song_index = 0
27       filled_face = []
28   
29       # Replace parts of the outline with song titles
30       for line in outline.splitlines():
31           if ' ' * 5 in line:  # Find lines with empty space to replace with songs
32               if song_index < len(song_list):
33                   # Insert song name to replace space in line
34                   filled_line = line.replace(' ' * 5, song_list[song_index].center(10))  # Center the text
35                   filled_face.append(filled_line)
36                   song_index += 1
37               else:
38                   filled_face.append(line)
39           else:
40               filled_face.append(line)
41   
42       return "\n".join(filled_face)
