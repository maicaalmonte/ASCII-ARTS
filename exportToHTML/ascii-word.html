4    def caesar_cipher(text, shift):
5        result = []
6        for char in text:
7            if char.isalpha():
8                shift_amount = shift % 26
9                if char.islower():
10                   result.append(chr((ord(char) - ord('a') + shift_amount) % 26 + ord('a')))
11               elif char.isupper():
12                   result.append(chr((ord(char) - ord('A') + shift_amount) % 26 + ord('A')))
13           else:
14               result.append(char)  # Non-alphabetic characters remain unchanged
15       return ''.join(result)
