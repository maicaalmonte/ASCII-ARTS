1    from PIL import Image

3    # Function to convert image to grayscale
4    def grayscale_image(image_path):
5        img = Image.open(image_path)
6        img = img.convert("L")  # Convert to grayscale
7        return img
8    
9    # Function to map pixel intensity to ASCII characters
10   def pixel_to_ascii(pixel_value):
11       ascii_chars = ["s", "n", "o", "o", "p", "y"]
12       return ascii_chars[pixel_value // 25]
13   
14   # Function to convert an image to ASCII art
15   def image_to_ascii(image_path, new_width=80):
16       img = grayscale_image(image_path)
17       width, height = img.size
18       aspect_ratio = height / width
19       new_height = int(aspect_ratio * new_width)
20   
21       # Resize the image
22       img = img.resize((new_width, new_height))
23   
24       ascii_str = ""
25       for y in range(new_height):
26           for x in range(new_width):
27               pixel_value = img.getpixel((x, y))
28               ascii_str += pixel_to_ascii(pixel_value)
29           ascii_str += "\n"
30   
31       return ascii_str
32   
33   # Function to save ASCII art to a text file
34   def save_ascii_art(ascii_art, filename="ascii_art.txt"):
35       with open(filename, "w") as f:
36           f.write(ascii_art)
37   
38   # Main function to generate and save the ASCII art
39   def main(image_path):
40       ascii_art = image_to_ascii(image_path)
41       save_ascii_art(ascii_art)
42       print("ASCII art generated and saved to ascii_art.txt")
